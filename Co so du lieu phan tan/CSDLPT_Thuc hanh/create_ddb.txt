/*
khoa1(makhoa,tenkhoa)
khoa2(makhoa,tenkhoa)

bomon1(mabm, tenbm, makhoa)
bomon2(mabm, tenbm, makhoa)

giangvien1(magv, hoten, phai, tuoi, mabm)
giangvien2(magv, hoten, phai, tuoi, mabm)

monhoc1(mamh, tenmh, mabm)
monhoc2(mamh, tenmh, mabm)

day(magv, mamh)
*/

create table khoa1
as
select *
from khoa 
where makhoa = 1;

create table khoa2
as
select *
from khoa 
where makhoa = 2;

create table bomon1
as
select mabm, tenbm, bomon.makhoa
from bomon, khoa1
where bomon.makhoa = khoa1.makhoa;

create table bomon2
as
select mabm, tenbm, bomon.makhoa
from bomon, khoa2
where bomon.makhoa = khoa2.makhoa;


create table giangvien1
as
select magv, hoten, phai, tuoi, giangvien.mabm
from giangvien, bomon1
where giangvien.mabm = bomon1.mabm;

create table giangvien2
as
select magv, hoten, phai, tuoi, giangvien.mabm
from giangvien, bomon2
where giangvien.mabm = bomon2.mabm;

create table monhoc1
as
select mamh, tenmh, monhoc.mabm
from monhoc, bomon1
where monhoc.mabm = bomon1.mabm;

create table monhoc2
as
select mamh, tenmh, monhoc.mabm
from monhoc, bomon2
where monhoc.mabm = bomon2.mabm;
