variable ma varchar2(30)
begin
  delete from ketqua;
  select hoten into :ma from sinhvien where masv = '3';
exception
  when no_data_found then
    begin
      select hoten into :ma from sinhvien where masv = '4';
    exception
      when no_data_found then
        :ma := 'khong co';
    end;    
end;
/
print :ma
