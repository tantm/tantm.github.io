Cơ sở dữ liệu tập trung

select mamh,tenmh
from monhoc
where mabm=$mabm and
	  mamh not in (select mamh
				   from day)

Cơ sở dữ liệu phân tán


select 1
from bmcntt
where mabm=$mabm
if #found
{
	select mamh,tenmh
	from mhcntt
	where mamh not in (select mamh
					   from daycntt)
}
else
{
	select 1
	from bmmnon
	where mabm=$mabm
	if #found
	{
		select mamh,tenmh
		from mhmnon
		where mamh not in (select mamh
						   from daymnon)
	}
}
